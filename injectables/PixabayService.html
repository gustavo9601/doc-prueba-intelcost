import {Injectable} from '@angular/core';

/**
 * Service- Servicio consumo API Pixabay
 */

import {environment} from '../../environments/environment';
import {HttpClient} from '@angular/common/http';
import {HitsInterface} from '../interfaces/hits.interface';

@Injectable({
  providedIn: 'root'
})
export class PixabayService {

  url: string = environment.url;
  key: string = environment.key;
  public perPage: number = 9;
  public category = null;
  public query = null;

  public categories = [
    'science',
    'education',
    'people',
    'feelings',
    'computer',
    'buildings'];


  constructor(private httpClient: HttpClient) {
  }

  /**
   * @param {number} page  recibe el numero de pagina
   * @returns {<HitsInterface>} Consume el api y devuelve el resultado para el front page
   */
  getImagesByParams(page: number = 0) {
    let urlPixabay = this.url + this.key + '&per_page=' + this.perPage;
    if (page > 0) {
      urlPixabay += '&page=' + page;
    }
    if (this.category) {
      urlPixabay += '&category=' + this.category;
    }
    if (this.query) {
      urlPixabay += '&q=' + this.query;
    }
    return this.httpClient.get<HitsInterface>(urlPixabay);
  }

  /**
   * @param {number} id  recibe el id de la imagen
   * @returns {<HitsInterface>} Consume el api y devuelve el resultado para el detail-product
   */
  getImageById(id: number) {
    let urlPixabay = this.url + this.key + '&id=' + id;
    return this.httpClient.get<HitsInterface>(urlPixabay);
  }

}
