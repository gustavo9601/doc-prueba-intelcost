import {Component, OnInit} from '@angular/core';
import {ActivatedRoute, ParamMap, Router} from '@angular/router';
import {PixabayService} from '../../services/pixabay.service';
import {HitInterface, HitsInterface} from '../../interfaces/hits.interface';
import {Location} from '@angular/common';

/**
 * Detail product - Detalle de la imagen o producto
 */

@Component({
  selector: 'app-detail-product',
  templateUrl: './detail-product.component.html',
  styleUrls: ['./detail-product.component.css']
})
export class DetailProductComponent implements OnInit {

  idImage: number = 0;
  image: HitInterface = null;
  tags: string[] = [];

  constructor(public _pixabayService: PixabayService,
              private activatedRoute: ActivatedRoute,
              private router: Router,
              private location: Location) {
  }

  /**
   * @returns Init
   */
  ngOnInit(): void {
    this.getIdImageUrl();
    this.getInfoImageById();
  }

  /**
   * @returns Captura id pasado por url
   */
  getIdImageUrl() {
    this.activatedRoute.paramMap.subscribe(
      (params: ParamMap) => {
        const idImage = parseInt(params.get('id'));
        if (Number.isNaN(idImage)) {
          this.router.navigate(['/home']);
        } else {
          this.idImage = idImage;
        }
      }
    );
  }
  /**
   * @returns Consume el servicio para obtener la informacion de la imagen por id
   */
  getInfoImageById() {
    this._pixabayService.getImageById(this.idImage).subscribe(
      (response: HitsInterface) => {
        if (response.totalHits === 1) {
          this.image = response.hits[0];
          this.tags = this.image.tags.split(',');
        } else {
          this.router.navigate(['/home']);
        }
      }, (error) => {
        this.router.navigate(['/home']);
      }
    );
  }

  /**e
   * @returns Retorna a la pagina anterior
   */
  backPage() {
    this.location.back();
  }

}
