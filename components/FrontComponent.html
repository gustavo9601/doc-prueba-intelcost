import {Component, OnInit} from '@angular/core';
import {PixabayService} from '../../services/pixabay.service';
import {HitInterface, HitsInterface} from '../../interfaces/hits.interface';
import {ActivatedRoute, ParamMap, Router} from '@angular/router';

/**
 * Front page - Pagina Incial
 */

@Component({
  selector: 'app-front',
  templateUrl: './front.component.html',
  styleUrls: ['./front.component.css']
})
export class FrontComponent implements OnInit {

  images: HitInterface[] = [];
  cantidadPages = [];
  currentPage = 1;
  category = null;
  query = null;
  showDropDown: boolean = false;
  loading: boolean = false;
  notFound: boolean = false;

  constructor(public _pixabayService: PixabayService,
              private activatedRoute: ActivatedRoute,
              private router: Router) {
  }

  /**
   * @returns Init
   */

  ngOnInit(): void {
    this.getNumberPageUrl();
    this.query = this._pixabayService.query;
    this.category = this._pixabayService.category;
  }

  /**
   * @returns Captura los filtros, para enviar el query
   */
  getImagesParams() {
    this.loading = true;
    this._pixabayService.getImagesByParams(this.currentPage).subscribe(
      (response: HitsInterface) => {
        if (response.total > 0) {
          this.notFound = false;
          this.images = response.hits;
          this.cantidadPages = new Array(Math.round(response.totalHits / this._pixabayService.perPage));
        } else {
          this.images = [];
          this.cantidadPages = [];
          this.notFound = true;
        }
        this.loading = false;
      }
    );
  }

  /**
   * @returns Obtiene si hay numero de pagina en la url
   */
  getNumberPageUrl() {
    this.activatedRoute.paramMap.subscribe(
      (params: ParamMap) => {
        const numberPage = parseInt(params.get('page'));
        if (Number.isNaN(numberPage)) {
          this.router.navigate(['/home']);
        } else {
          this.currentPage = numberPage;
        }
        this.getImagesParams();
      }
    );
  }

  /**
   * @returns Seteo de variables en el servicio
   */
  getResultsQuery() {
    this._pixabayService.query = this.query;
    this.getImagesParams();
    this.router.navigate(['/home']);
  }

  /**
   * @returns Reset y seteo de variables Category
   */
  filterCategory(category) {
    this.showDropDown = false;
    this.images = [];
    this.cantidadPages = [];
    this.category = category;
    this._pixabayService.category = this.category;
  }

}
